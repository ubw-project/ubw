exports.createAddress = function() {
    return new Promise(function(resolve, reject) {
        try {
            var keyPair = bitcoinjs.ECPair.makeRandom();
            resolve({address: keyPair.getAddress(), key: keyPair.toWIF()});
        } catch(exception) {
            reject(exception)
        }
    });
}
